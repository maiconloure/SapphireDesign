## Install

```bash
npm install --save capstone
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'capstone'
import 'capstone/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
